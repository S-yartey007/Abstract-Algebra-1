\section{Group Structure on $G/\sim$}

\textbf{Proof:} \\
We have already noted that the stated condition is necessary. [cite: 2] To prove it is sufficient, with the stated consequences, assume $a, a', g \in G$: [cite: 3]
\begin{equation*}
    a \sim a' \Rightarrow ga \sim ga' \text{ and } ag \sim a'g \text{ [cite: 4]}
\end{equation*}

Then the operation $[g] \cdot [b] := [gb]$ [cite: 6] is well-defined and we have to verify that it defines a group structure on $G/\sim$. [cite: 7, 8]

\begin{itemize}
    \item \textbf{Associativity:} For all $a, b, c \in G$, [cite: 9] associativity is inherited from the group $G$: [cite: 10]
    \[ ([a] \cdot [b]) \cdot [c] = [ab] \cdot [c] = [(ab)c] = [a(bc)] = [a] \cdot [bc] = [a] \cdot ([b] \cdot [c]) \text{ [cite: 11, 12, 13]} \]
    \item \textbf{Identity:} The class $[e_G]$ is an identity with respect to this operation: $\forall g \in G$ [cite: 14]
    \[ [g] \cdot [e_G] = [ge_G] = [g], \quad [e_G] \cdot [g] = [e_Gg] = [g] \text{ [cite: 15]} \]
    \item \textbf{Inverse:} The class $[g^{-1}]$ is the inverse of $[g]$: [cite: 16]
    \[ [g^{-1}] \cdot [g] = [g^{-1}g] = [e_G], \quad [g] \cdot [g^{-1}] = [gg^{-1}] = [e_G] \text{ [cite: 17]} \]
\end{itemize}
Thus $G/\sim$ is indeed a group.

\section{Cosets}

In Proposition 4.3, we obtained the conditions: [cite: 21, 22]
\begin{enumerate}
    \item[(*)] $(\forall g \in G): a \sim b \Rightarrow ga \sim gb$ [cite: 23]
    \item[(**)] $(\forall g \in G): a \sim b \Rightarrow ag \sim bg$ [cite: 25]
\end{enumerate}
These lead to a complete description of all compatible relations on a group $G$ (where compatible means the equivalence relation respects the group operation). [cite: 30] Note that we have a group structure on $G/\sim$ only if both conditions are satisfied. [cite: 32]

\subsection{Description of (*)}

\textbf{Proposition 4.4:} Let $\sim$ be an equivalence relation on a group $G$ satisfying (*). [cite: 34, 37] Then:
\begin{itemize}
    \item The equivalence class of $e_G$ is a subgroup $H$ of $G$. [cite: 38]
    \item $a \sim b \iff a^{-1}b \in H \iff aH = bH$. [cite: 38, 51]
\end{itemize}

\textbf{Proof:}\\
Let $H \subseteq G$ be the equivalence class of the identity. $H \neq \emptyset$ as $e_G \in H$. [cite: 40, 42] For $a, b \in H$, we have $e_G \sim a$ and $e_G \sim b$, hence $b^{-1} \sim e_G$. [cite: 41, 42] Thus $ab^{-1} \sim a \sim e_G$ by transitivity, showing $H$ is a subgroup. [cite: 44, 45, 46]

Next, assume $a, b \in G$ and $a \sim b$. Multiplying on the left by $a^{-1}$ implies $e_G \sim a^{-1}b$, that is $a^{-1}b \in H$. [cite: 47] This implies $aH = bH$. [cite: 48, 51] Conversely, if $aH = bH$, then $a \in bH$, so $a^{-1}b \in H$, meaning $e_G \sim a^{-1}b$, which gives $a \sim b$. [cite: 54, 57, 58]

\textbf{Definition 4.5:} The \textbf{left-cosets} of a subgroup $H$ in $G$ are the sets $aH$ for $a \in G$. The \textbf{right-cosets} of $H$ are the sets $Ha$ for $a \in G$. [cite: 68]

\textbf{Proposition 4.6:} If $H$ is any subgroup of a group $G$, the relation $\sim_L$ defined by: [cite: 71, 73]
\[ \forall a, b \in G: a \sim_L b \iff a^{-1}b \in H \text{ [cite: 74, 75]} \]
is an equivalence relation satisfying (*). [cite: 76]

\section{Correspondence}

\textbf{Proposition 4.7:} There is a one-to-one correspondence between subgroups of $G$ and equivalence relations on $G$ satisfying (*). [cite: 83, 84, 85, 86] For the relation $\sim_L$ corresponding to $H$, $G/\sim_L$ is the set of left-cosets $aH$. [cite: 87, 88]

\textbf{Proposition 4.8:} There is a one-to-one correspondence between subgroups of $G$ and equivalence relations on $G$ satisfying (**). [cite: 91, 92, 93] For the relation $\sim_R$ corresponding to $H$, $G/\sim_R$ is the set of right-cosets $Ha$. [cite: 94, 95, 96]
\[ a \sim_R b \iff ab^{-1} \in H \iff Ha = Hb \text{ [cite: 99]} \]

\section{Example 4.9}

Let $G = S_3$, and let $H$ be the subgroup:
\[ H = \left\{ \begin{pmatrix} 1 & 2 & 3 \\ 1 & 2 & 3 \end{pmatrix}, \begin{pmatrix} 1 & 2 & 3 \\ 2 & 1 & 3 \end{pmatrix} \right\} \text{ [cite: 116]} \]
Consider the element $\begin{pmatrix} 1 & 2 & 3 \\ 3 & 1 & 2 \end{pmatrix}$: [cite: 117]
\begin{align*}
    \text{Left coset: } \begin{pmatrix} 1 & 2 & 3 \\ 3 & 1 & 2 \end{pmatrix} H &= \left\{ \begin{pmatrix} 1 & 2 & 3 \\ 3 & 1 & 2 \end{pmatrix}, \begin{pmatrix} 1 & 2 & 3 \\ 3 & 2 & 1 \end{pmatrix} \right\} \text{ [cite: 117]} \\
    \text{Right coset: } H \begin{pmatrix} 1 & 2 & 3 \\ 3 & 1 & 2 \end{pmatrix} &= \left\{ \begin{pmatrix} 1 & 2 & 3 \\ 3 & 1 & 2 \end{pmatrix}, \begin{pmatrix} 1 & 2 & 3 \\ 1 & 3 & 2 \end{pmatrix} \right\} \text{ [cite: 118]}
\end{align*}
This reflects the fact that the conditions (*) and (**) are different. [cite: 119, 120]
